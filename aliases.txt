# Zsh Aliases
# Copy this file to your cloud storage (e.g., mycloud/system/aliases.txt)
# Then add to CUSTOM_SYMLINKS in .zshrc:
#   "$CLOUD_FOLDER/system/aliases.txt|~/.aliases"
# Aliases will be auto-sourced after bootstrap setup.

# Navigation
alias .='cd ~'
alias ..='cd ..'
alias ...='cd ../..'
alias ....='cd ../../..'
alias ~='cd ~'
alias -- -='cd -'

# Listing
alias ls='ls -lh'
alias l='ls -lh'
alias la='ls -lAh'
alias ld='ls -lah | grep "^d"'
alias lf='ls -lah | grep "^-"'
alias ll='ls -lah | grep "^l"'
alias lh='ls -ldh .*'
alias l.='ls -ldh .*'

# Utilities
alias c='clear'
alias r='clear && exec zsh'
alias x='exit'
alias cp='cp -v'
alias mv='mv -v'
alias rm='rm -v'
alias mkdir='mkdir -p'

# Text Editors
alias vi='vim'
alias ea='vim ~/.zshrc'
alias ez='vim ~/.zshrc'

# SSH & Keys
alias pubkey="more ~/.ssh/id_rsa.pub | pbcopy | echo '=> Public key copied to pasteboard.'"
alias sshkey='cat ~/.ssh/id_ed25519.pub | pbcopy && echo "SSH key copied to pasteboard"'

# Git
alias g='git'
alias ga='git add'
alias gb='git branch'
alias gc='git commit'
alias gco='git checkout'
alias gd='git diff'
alias gf='git fetch'
alias gl='git log --oneline -10'
alias gp='git push'
alias gpull='git pull'
alias gs='git status'

# Network
alias ping='ping -c 5'
alias ip='curl http://ipecho.net/plain; echo'
alias ports='lsof -i -P -n | grep LISTEN'

# System
alias df='df -h'
alias du='du -h'
alias top='top -o cpu'
alias ps='ps aux'
alias memhog='ps aux | sort -nk 3 | tail'
